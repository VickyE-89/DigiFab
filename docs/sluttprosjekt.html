Sluttprosjektet

Som mitt slutt prosjekt ønsket jeg å lage et videreutviklet, fullverdig reaksjonsspill. Jeg fikk litt blod på tann ved forrige prosjekt, Raspberry Pi prosjektet mitt. Jeg ønsker å lage et full size spill, fire spiller, som skal stå på Makerspace. Tanken og ønsket er at fremmede mennesker skal kunne stoppe på Makerspace og ta et raskt spill sammen før man går videre til sitt. Smittevern er selvfølgelig også tatt med i betraktning. Tanken er egne knapper og stort nok brett til å kunne holde en meters avstand under hele spillet. Når man er ferdig vasker man enkelt knappene.

Så, for dette prosjektet har jeg laget en prototype av dette. Det er et én spiller spill, men konseptet er det samme. Når et lys lyser skal man trykke på knappen med samme farge. La oss ta en titt på hva jeg har gjort.

Jeg startet med å gjøre litt forskjellig testing. Jeg testet knapper og lys opp i mot hverandre og koblet knappene i en resistor ladder. I sluttresultatet gikk jeg vekk fra dette og kablet hver knapp til hver sin pin. Dette fordi jeg ikke fikk koden helt til å fungere når jeg skulle bruke forskjellige verdier på samme pin. Med litt bedre tid, og for videreutvikling av dette igjen, kommer jeg til å gå tilbake til å koble hver spillers knapper til hver sin resistor ladder. Jeg anser det som det mest effektive å gjøre. Før vi ser nærmere på byggeprosessen, ønsker jeg å vise koden for programmet. For referanse til koden får du her en pin oversikt.

La oss ta en nærmere titt på koden. Jeg har gjort mye undersøkelser og finner ingen som har gjort det akkuratt slik som jeg har. De fleste reaksjonspillene jeg har funnet på nett innebærer at man skal være første mann til å trykke på de samme knappene, eller at man har én knapp man skal trykke på når et bestemt lys lyser. Jeg har imidlertid funnet inspirasjon fra en fyr på YouTube, Tutorial X, i tillegg til forelesnings videoer i faget. Link til YouTube videoen: Arduino Tutorial: 13. LED Reaction Game - part 2. Jeg har brukt "attachInterrupt" for å få knappene til å avbryte programmet, og koden min ser slik ut:

//Definerer alle lampene til pin og setter som variabler for senere bruk
#define yellowLED 3
#define blueLED 4
#define redLED 5
#define greenLED 6
#define whiteLED 7

// Tilfeldig tider lysene kan lyse i
int timesOn[] = {400, 600, 800, 1000, 1200};

// Tilfeldig forsinkelse mellom lysene
int delayTimes[] = {500, 700, 900, 1100, 1300};

//Variabler for å sjekke om lysene er på eller ikke
volatile boolean yellowLight = false;
volatile boolean blueLight = false;
volatile boolean redLight = false;
volatile boolean greenLight = false;
volatile boolean whiteLight = false;

// For å sjekke om man har vunnet.
volatile boolean isWinner = false;

void setup() {
pinMode(yellowLED, OUTPUT); //Gult lys
pinMode(blueLED, OUTPUT); // Blått lys
pinMode(redLED, OUTPUT); // Rødt lys
pinMode(greenLED, OUTPUT); // Grønt lys
pinMode(whiteLED, OUTPUT); // Hvitt lys

attachInterrupt(digitalPinToInterrupt(2), checkWinnerYellow, CHANGE); // Gul knapp
attachInterrupt(digitalPinToInterrupt(18), checkWinnerBlue, CHANGE); // Blå knapp
attachInterrupt(digitalPinToInterrupt(19), checkWinnerRed, CHANGE); // Rød knapp
attachInterrupt(digitalPinToInterrupt(20), checkWinnerGreen, CHANGE); // Grønn knapp
attachInterrupt(digitalPinToInterrupt(21), checkWinnerWhite, CHANGE); // Hvit knapp
turnOffLEDs();
}

void loop() {
int randomLED = random(1,6); //Velger et tilfeldig lys
int timeLEDIsLit = timesOn[random(5)]; //Velger en tilfeldig tid lyset skal være tent fra listen(array) definert over
lightLED(randomLED, timeLEDIsLit); //Skrur på det valgte lyset
turnOffLEDs(); //Skrur av alle lys
int delayTime = delayTimes[random(5)]; //pauser i tilfeldig valgt tid fra listen definert over
if (isWinner) { //Sjekker om vinner variabelen er sann
isWinner = false; //Hvis det er en vinner, setter vinner variabelen tilbake til falsk
winner(); //og deretter utfører vinner sekvensen
}
}

//Metode for å skru på tilfeldig lys
//Bruker switch-case for å skille de forskjellige mulighetene
//Hvert tilfelle skrur på hvert sitt lys og setter tilhørende variabel til sann
void lightLED(int led, int duration) {
switch (led) {
case 1: // Gul
digitalWrite(yellowLED, HIGH);
yellowLight = true;
break;
case 2: //Blå
digitalWrite(blueLED, HIGH);
blueLight = true;
break;
case 3: // Rød
digitalWrite(redLED, HIGH);
redLight = true;
break;
case 4: // Grønn
digitalWrite(greenLED, HIGH);
greenLight = true;
break;
case 5: // Hvit
digitalWrite(whiteLED, HIGH);
whiteLight = true;
break;
default:
break;
}
//vent forsinkelses tiden
delay(duration);
// Nullstiller alle variabler for påskrudd lys
yellowLight = false;
blueLight = false;
redLight = false;
greenLight = false;
whiteLight = false;
}

//Metode for å skru alle lys av
void turnOffLEDs() {
digitalWrite(yellowLED, LOW);
digitalWrite(blueLED, LOW);
digitalWrite(redLED, LOW);
digitalWrite(greenLED, LOW);
digitalWrite(whiteLED, LOW);
}

//Metode for å indikere et vinn
void winner() {
// Alle lys blinker 3 ganger hver gang man trykker på riktig knapp til riktig tid
for (int i=0; i<3; i++) {
digitalWrite(3, HIGH);
digitalWrite(4, HIGH);
digitalWrite(5, HIGH);
digitalWrite(6, HIGH);
digitalWrite(7, HIGH);
delay(250);
digitalWrite(3, LOW);
digitalWrite(4, LOW);
digitalWrite(5, LOW);
digitalWrite(6, LOW);
digitalWrite(7, LOW);
delay(250);
}
delay(2000);
}

//Metode kalt når gul knapp trykkes
void checkWinnerYellow() {
if(yellowLight) { //Hvis variabel for gult lys er sann
isWinner = true; // Sett variabel for vinner til sann
}
}

//Metode kalt når blå knapp trykkes
void checkWinnerBlue() {
if(blueLight) { //Hvis variabel for blått lys er sann
isWinner = true; //Sett variabel for vinner til sann
}
}

//Metode kalt når rød knapp trykkes
void checkWinnerRed() {
if(redLight) { //Hvis variabel for rødt lys er sann
isWinner = true; //Sett variabel for vinner til sann
}
}

//Metode kalt når grønn knapp trykkes
void checkWinnerGreen() {
if(greenLight) { //Hvis variabel for grønt lys er sann
isWinner = true; //Sett variabel for vinner til sann
}
}

//Metode kalt når hvit knapp trykkes
void checkWinnerWhite() {
if(whiteLight) { //Hvis variabel for hvitt lys er sann
isWinner = true; //Sett variabel for vinner til sann
}
}
